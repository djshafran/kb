def avg_color(colors):
    r, g, b = 0, 0, 0

    for color in colors:
        r += int(color[1:3], 16)
        g += int(color[3:5], 16)
        b += int(color[5:7], 16)

    r //= len(colors)
    g //= len(colors)
    b //= len(colors)

    return f"#{r:02X}{g:02X}{b:02X}"

def main():
